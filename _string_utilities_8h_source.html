335 class CompareSubString
336 {
337  public:
338  Boolean operator()(const SubString &a, const SubString &b) const {
339  Int32 aSize = a.Length();
340  Int32 bSize = b.Length();
341  if (aSize != bSize) {
342  return aSize < bSize;
343  } else {
344  const Utf8Char* ba = a.Begin();
345  const Utf8Char* bb = b.Begin();
346  const Utf8Char* ea = a.End();
347 
348  while (ba < ea) {
349  if (*ba != *bb) {
350  return *ba < *bb;
351  } else {
352  ba++;
353  bb++;
354  }
355  }
356  return false;
357  }
358  }
359 };
